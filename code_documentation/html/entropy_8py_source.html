57 def Shannon(data):
58  '''
59  Calculates Shannon entropy.
60 
61  @param data: data for entropy calculation
62  @type data: string or numerical values or list
63  @return: Shanon entropy of data
64  @rtype: float
65  '''
66  (data, length) = _process_data(data)
67  result = [data[k]/length for k in data.iterunique()]
68  result = [x * math.log(x, 2) for x in result]
69  result = (-1) * sum(result)
70  return result

89 def Renyi(data, order=2):
90  '''
91  Calculates Renyi entropy, which is a generalized version of Shannon
92  entropy. When order = 1, Renyi entropy = Shannon entropy,
93 
94  @param data: data for entropy calculation
95  @type data: string or numerical values or list
96  @param order: exponent
97  @type order: integer
98  @return: Renyi entropy of data
99  @rtype: float
100  '''
101  (data, length) = _process_data(data)
102  order = int(order)
103  if order < 0:
104  raise FunctionParameterValueError('order must be at least zero')
105  if order == 1:
106  return Shannon(data)
107  else:
108  result = [(data[k]/length) ** order for k in data.iterunique()]
109  result = (1/(1-order)) * math.log(sum(result), 2)
110  return result
