283  def extractGreedySeriesValue(self, series_names, operator, value,
284  new_dataframe_name=''):
285  '''!
286  Method for "greedy" extraction of series name(s) and value by
287  the following:
288  1. Generate a new data frame by extracting required series
289  using Dataframe.extractSeries method, which is essentially
290  column reduction.
291  2. Reduce the data labels (essentially, row reduction) by
292  finding data values in any remaining (one or more) series
293  using the search criterion.
294 
295  This method is considered to be "greedy" as the row reduction
296  is not specific to particular series (column).
297 
298  @param series_names list: names of series to extract
299  @param operator: comparative operator. Allowed values are: '>'
300  (more than), '<' (less than), '>=' (more than or equals to),
301  '<=' (less than or equals to), '=' (equals to), '!=' (not
302  equals to), and '*' (all, basically replicating the entire
303  data frame).
304  @param value: value of the data to compare.
305  @param new_dataframe_name string: name for new data frame (
306  that is to be returned)
307  @type new_dataframe_name: string
308  @return dataframe.Dataframe object
309  '''
310  df = self.extractSeries(series_names, new_dataframe_name)
311  return df.extractValue(operator, value, new_dataframe_name)

313  def extractSeriesValue(self, series_name, operator, value,
314  new_dataframe_name=''):
315  '''!
316  Method for extraction of row data where a specified value or
317  range of value is found in the current data frame.
318 
319  This method is logically identical to SQL select.
320 
321  select * from <current> where <current>.seriesA > 30
322 
323  can be represented as
324 
325  >>> df = <current>.extractSeriesValue('seriesA', '>', 30, '')
326 
327  @param series_name list: name of series to extract
328  @param operator: comparative operator. Allowed values are: '>'
329  (more than), '<' (less than), '>=' (more than or equals to),
330  '<=' (less than or equals to), '=' (equals to), '!=' (not
331  equals to), and '*' (all, basically replicating the entire
332  data frame).
333  @param value: value of the data to compare.
334  @param new_dataframe_name string: name for new data frame (
335  that is to be returned)
336  @return dataframe.Dataframe object
337  '''
338  df = Dataframe(new_dataframe_name)
339  try:
340  data = {}
341  index = self.series_names.index(series_name)
342  for label in self.data.keys():
343  if operator == '=' and self.data[label][index] == value:
344  data[label] = [x for x in self.data[label]]
345  elif operator == '>' and self.data[label][index] > value:
346  data[label] = [x for x in self.data[label]]
347  elif operator == '<' and self.data[label][index] < value:
348  data[label] = [x for x in self.data[label]]
349  elif operator == '>=' and self.data[label][index] >= value:
350  data[label] = [x for x in self.data[label]]
351  elif operator == '<=' and self.data[label][index] <= value:
352  data[label] = [x for x in self.data[label]]
353  elif operator == '*':
354  data[label] = [x for x in self.data[label]]
355  df.data = data
356  df.series_names = [name for name in self.series_names]
357  df.label = data.keys()
358  except IOError: pass
359  return df

381  def extractValue(self, operator, value, new_dataframe_name=''):
382  '''!
383  Method to extract one or more data labels across all series,
384  based on criterion, from the current data frame into a new
385  data frame.
386 
387  For example, the following will extract all data labels across
388  all series where data value is more than 30, and generate and
389  return a new dataframe (ndf).
390 
391  >>> ndf = df.extractValue('>', 30, 'newframe')
392 
393  @param operator: comparative operator. Allowed values are: '>'
394  (more than), '<' (less than), '>=' (more than or equals to),
395  '<=' (less than or equals to), '=' (equals to), '!=' (not
396  equals to), and '*' (all, basically replicating the entire
397  data frame).
398  @param value: value of the data to compare.
399  @param new_dataframe_name string: name for new data frame (
400  that is to be returned)
401  @return dataframe.Dataframe object
402  '''
403  df = Dataframe(str(new_dataframe_name))
404  data = {}
405  for label in self.data.keys():
406  try:
407  if (operator == '>') and \
408  sum([1 for item in self.data[label] if item > value]):
409  data[label] = [x for x in self.data[label]]
410  elif (operator == '<') and \
411  sum([1 for item in self.data[label] if item < value]):
412  data[label] = [x for x in self.data[label]]
413  elif (operator == '>=') and \
414  sum([1 for item in self.data[label] if item >= value]):
415  data[label] = [x for x in self.data[label]]
416  elif (operator == '<=') and \
417  sum([1 for item in self.data[label] if item <= value]):
418  data[label] = [x for x in self.data[label]]
419  elif (operator == '=') and \
420  sum([1 for item in self.data[label] if item == value]):
421  data[label] = [x for x in self.data[label]]
422  elif (operator == '!=') and \
423  sum([1 for item in self.data[label] if item != value]):
424  data[label] = [x for x in self.data[label]]
425  elif (operator == '*'):
426  data[label] = [x for x in self.data[label]]
427  except KeyError: pass
428  df.data = data
429  df.label = data.keys()
430  df.series_names = [x for x in self.series_names]
431  return df

447  '''!
448  Method to add a data series into the data frame.
449 
450  @param series: data series (dataframe.Series object) for
451  addition.
452  @param fill_in: value to fill into missing values during
453  process. This is required as the number of data elements
454  across each label must be the same. Hence, filling in of
455  missing values can occur when (1) the newly added data series

465  df_label = self.data.keys()
466  for i in range(len(series.data)):

468  temp = [fill_in] * len(self.series_names)
469  temp.append(series.data[i])
470  self.data[series.label[i]] = temp
471  else:
472  temp = self.data[series.label[i]]
473  temp.append(series.data[i])
474  self.data[series.label[i]] = temp
475  self.series_names.append(series.name)
476  self.label = self.data.keys()
477  for k in self.label:

483  def addData(self, dataset, labels, fill_in=None):
484  '''!
485  Method to add new data into the current data frame. For
486  example,
487 
488  >>> df = d.Dataframe('frame1')
489  >>> dataset = {'seriesA': [10, 11, 12, 13, 14],
490  'seriesB': [20, 21, 22, 23, 24],
491  'seriesC': [30, 31, 32, 33, 34],
492  'seriesD': [40, 41, 42, 43, 44]}
493  >>> label = ['A', 'B', 'C', 'D', 'E']
494  >>> df.addData(dataset, label)
495 
496  will result in::
497 
498  seriesA seriesB seriesC seriesD
499  A 10 20 30 40
500  B 11 21 31 41
501  C 12 22 32 42
502  D 13 23 33 43
503  E 14 24 34 44
504 
505  @param dataset dictionary: set of data to add. This is
506  formatted as a dictionary where the key is the series name and
507  the value is a list of data values of the same number of
508  elements as labels.
509  @param labels list: list of labels for the data values. If not
510  given, a sequential number will be given as label but this
511  does not ensure uniqueness in label names across the entire
512  series.
513  @param fill_in: value to fill into missing values during
514  process. This is required as the number of data elements
515  across each label must be the same. Hence, filling in of
516  missing values can occur when (1) the newly added data series
517  consists of new labels which are not found in the current data
518  frame (this will require filling in of missing values in the
519  current data frame), or (2) the current data frame consists of
520  labels that are not found in the newly added data series (this
521  will require filling in of missing values to the newly added
522  data series). Default = None.
523  '''
524  series_names = dataset.keys()
525  series_names.sort()
526  for series_name in series_names:
527  s = Series(str(series_name))
528  s.addData(dataset[series_name], labels)
529  self.addSeries(s, fill_in)
