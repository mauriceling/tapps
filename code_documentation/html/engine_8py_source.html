55 def LoadPlugin(session, plugin):
56  '''!
57  Function to perform basic checks and load a plugin into session
58  dictionary to get it ready for use.
59 
60  The following checks are done:
61  1. Presence of callable main function in plugin
62  2. Presence of Parameters dictionary
63  3. Presence of instructions
64  4. Presence of plugin name
65  5. Presence of release number
66  6. Presence of acceptable category
67  7. Presence of short description
68  8. Presence of long description
69  9. Presence of URL
70  10. Presence of contact details
71  11. Presence of license
72 
73  The following changes will be made to session dictionary:
74  1. If plugin is successfully loaded, the plugin name will be
75  appended to session['plugins']['loaded']
76  2. If plugin is successfully loaded, the plugin name will be
77  appended to session['plugins'][<plugin category>]
78  3. If plugin is successfully loaded, details of the plugin
79  will be loaded into session['plugins_<plugin name>'], and
80  session['plugins_<plugin name>']['main'] will contain the
81  function entry point for the plugin
82  4. If plugin is NOT successfully loaded (failure in one or
83  more of the above checklist), the checklist for the plugin
84  will be loaded into session['plugins']['loadFail'][<plugin
85  name>]
86 
87  @param session: dictionary to hold all data within the current
88  session. Please see module documentation for more details.
89  @param plugin String: module name of plugin to load (corresponding
90  to the folder/dictionary which the plugin resides - <current
91  working directory>/plugin/<plugin folder name>)
92  @return: session dictionary
93  '''
94  checks = ['MainFunctionError',
95  'ParametersError',
96  'InstructionsError',
97  'NameError',
98  'ReleaseError',
99  'CategoryError',
100  'ShortDescriptionError',
101  'LongDescriptionError',
102  'URLError',
103  'ContactError',
104  'LicenseError']
105  # Check 1: Presence of callable main function in plugin
106  try:
107  if callable(plugin.main):
108  main = plugin.main
109  checks[0] = 'Passed'
110  else:
111  print('MainFunctionError - main not callable')
112  except:
113  print('MainFunctionError - general exception')
114  # Check 2: Presence of Parameters dictionary
115  try:
116  if type(plugin.parameters) == type({}):
117  parameters = plugin.parameters
118  checks[1] = 'Passed'
119  else:
120  print('ParametersError - parameters type is not dictionary')
121  except:
122  print('ParametersError - general exception')
123  # Check 3: Presence of instructions
124  try:
125  instructions = plugin.instructions
126  checks[2] = 'Passed'
127  except: print('InstructionsError')
128  # Check 4: Presence of plugin name
129  try:
130  plugin_name = plugin.name
131  checks[3] = 'Passed'
132  except: print('NameError')
133  # Check 5: Presence of release number
134  try:
135  release = plugin.release
136  checks[4] = 'Passed'
137  except: print('ReleaseError')
138  # Check 6: Presence of acceptable category
139  try:
140  category = plugin.category
141  if category in plugin_categories:
142  checks[5] = 'Passed'
143  else:
144  print('CategoryError - Invalid category')
145  except: print('CategoryError - general exception')
146  # Check 7: Presence of short description
147  try:
148  sDesc = plugin.shortDescription
149  checks[6] = 'Passed'
150  except: print('ShortDescriptionError')
151  # Check 8: Presence of long description
152  try:
153  lDesc = plugin.longDescription
154  checks[7] = 'Passed'
155  except: print('LongDescriptionError')
156  # Check 9: Presence of URL
157  try:
158  URL = plugin.projectURL
159  checks[8] = 'Passed'
160  except: print('URLError')
161  # Check 10: Presence of contact details
162  try:
163  contact = plugin.contactDetails
164  checks[9] = 'Passed'
165  except: print('ContactError')
166  # Check 11: Presence of license
167  try:
168  license = plugin.license
169  checks[10] = 'Passed'
170  except: print('LicenseError')
171  # Tabulate passes
172  pass_rate = float(len([x for x in checks if x == 'Passed'])) / float(len(checks))
173  if pass_rate < 1.0:
174  session['plugins']['loadFail'][plugin] = checks
175  else:
176  plugin_name = plugin_name.lower()
177  parameters['results'] = Dataframe()
178  session['plugins']['loaded'].append(plugin_name)
179  session['plugins'][category].append(plugin_name)
180  session['plugin_' + plugin_name] = {'main': main,
181  'parameters': parameters,
182  'instructions': instructions,
183  'release': release,
184  'sdesc': sDesc,
185  'ldesc': lDesc,
186  'URL': URL,
187  'contact': contact,
188  'license': license}
189  return session

191 def GetPlugins(session, pluginpath='plugins'):
192  '''!
193  Function to discover available plugins and load each of the
194  plugins (using engine.LoadPlugin function to each plugin) into
195  session dictionary and get the plugins ready for use. Each plugin
196  is a Python script file (with or without its own folder) in the
197  plugin directory/folder; hence, the plugins discovery process is
198  to list down the directory/folder names within the plugin directory
199  /folder.
200 
201  As this function calls engine.LoadPlugin function repeatedly to
202  each plugin, the following changes will be made to session
203  dictionary:
204  1. If plugin is successfully loaded, the plugin name will be
205  appended to session['plugins']['loaded']
206  2. If plugin is successfully loaded, the plugin name will be
207  appended to session['plugins'][<plugin category>]
208  3. If plugin is successfully loaded, details of the plugin
209  will be loaded into session['plugins_<plugin name>'], and
210  session['plugins_<plugin name>']['main'] will contain the
211  function entry point for the plugin
212  4. If plugin is NOT successfully loaded (failure in one or
213  more of the above checklist), the checklist for the
214  plugin will be loaded into
215  session['plugins']['loadFail'][<plugin name>]
216 
217  @param session: dictionary to hold all data within the current
218  session. Please see module documentation for more details.
219  @param path String: path to plugin directory/folder. Default =
220  plugins
221  @return: session dictionary
222  '''
223  pluginpath = str(pluginpath)
224  pysearchre = re.compile('.py$', re.IGNORECASE)
225  pluginfiles = filter(pysearchre.search,
226  os.listdir(os.path.join(os.path.dirname(__file__), pluginpath)))
227  form_module = lambda fp: '.' + os.path.splitext(fp)[0]
228  pluginlist = [x for x in map(form_module, pluginfiles)]
229  pluginlist = [x for x in pluginlist
230  if not (x.startswith('.__') or
231  x.startswith('.wikipage_generator') or
232  x.startswith('.template'))]
233  # import parent module / namespace
234  importlib.import_module('plugins')
235  for plugin in pluginlist:
236  print('Loading plugin: %s' % str(plugin[1:]))
237  imported_plugin = importlib.import_module(plugin,
238  package="plugins")
239  session = LoadPlugin(session, imported_plugin)
240  for category in plugin_categories:
241  plugin_list = session['plugins'][category]
242  plugin_list = list(set(plugin_list))
243  session['plugins'][category] = plugin_list
244  return session

246 def RunPlugin(session, parameter_name):
247  '''!
248  Function to run/execute a plugin using the parameters (
249  sub-dictionary within session, under 'parameters' key) needed for
250  the plugin and returning the execution results into session
251  dictionary.
252 
253  The parameters sub-dictionary will need to contain values for the
254  following keys:
255  - analysis_name: user-given name for the analytical execution
256  of plugin
257  - plugin_name: name of plugin to execute
258  - dataframe: a dataframe object to act as data for use by the
259  plugin and any other plugin-specific parameters/options.
260 
261  After execution, analysis results will be returned as value to
262  'results' key in the parameters sub-dictionary.
263 
264  Hence, session['parameters'][<parameter_name>] will hold the
265  entire parameters sub-dictionary for the current plugin execution
266  and session['parameters'][<parameter_name>]['results'] will hold
267  the plugin output results.
268 
269  @param session: dictionary to hold all data within the current
270  session. Please see module documentation for more details.
271  @param parameter_name: name of parameter dictionary to hold all
272  the parameters needed to execute the plugin. Please see module
273  documentation for more details.
274  @return: session dictionary
275  '''
276  plugin_name = 'plugin_' + \
277  session['parameters'][parameter_name]['plugin_name']
278  parameters = session[plugin_name]['main'](session['parameters'][parameter_name])
279  session['parameters'][parameter_name] = parameters
280  return session

282 def LoadCSV(session, filepath, series_header, separator, fill_in, newline):
283  '''!
284  Function to load a comma-delimited (CSV) file as a data frame.
285 
286  The data frame will have the filepath as name, and will be loaded
287  into session dictionary under 'new_dataframe' key.
288 
289  @param session: dictionary to hold all data within the current
290  session. Please see module documentation for more details.
291  @param filepath String: path to CSV file.
292  @param series_header: boolean flag to denote whether the first row
293  in the CSV file contains the data header. It is highly recommended
294  that header is included in the CSV file.
295  @param separator: item separator within the CSV file.
296  @param fill_in: value to fill into missing values during process.
297  This is required as the number of data elements across each label
298  must be the same. Hence, filling in of missing values can occur
299  when (1) the newly added data series consists of new labels which
300  are not found in the current data frame (this will require filling
301  in of missing values in the current data frame), or (2) the
302  current data frame consists of labels that are not found in the
303  newly added data series (this will require filling in of missing
304  values to the newly added data series).
305  @param newline: character to denote new line or line feed in the
306  CSV file.
307  '''
308  dataframe = Dataframe(filepath)
309  dataframe.addCSV(filepath, series_header, separator, fill_in, newline)
310  session['new_dataframe'] = dataframe
311  return session
