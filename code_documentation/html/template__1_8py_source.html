115 def summarize_series(dataframe):
116  '''
117  Function to statistical summaries of each series in the dataframe.
118  '''
119  labels = dataframe.data.keys()
120  series = dataframe.series_names
121  statistics = {'summation': [],
122  'arithmetic_mean': [],
123  'standard_deviation': [],
124  'maximum': [],
125  'minimum': [],
126  'median': [],
127  'count': []}
128  for index in range(len(series)):
129  sdata = [float(dataframe.data[label][index]) for label in labels]
130  # 1. summation
131  statistics['summation'].append(sum(sdata))
132  # 2. arithmetic mean
133  arithmetic_mean = float(sum(sdata)) / len(sdata)
134  statistics['arithmetic_mean'].append(arithmetic_mean)
135  # 3. median
136  index = int(len(sdata) / 2)
137  statistics['median'].append(sdata[index])
138  # 4. maximum
139  sdata.sort()
140  statistics['maximum'].append(sdata[-1])
141  # 5. minimum
142  statistics['minimum'].append(sdata[0])
143  # 6. count
144  statistics['count'].append(len(sdata))
145  # 7. standard deviation
146  if len(sdata) < 30:
147  s = float(sum([(x-arithmetic_mean)**2
148  for x in sdata])) / (len(sdata) - 1)
149  else:
150  s = float(sum([(x-arithmetic_mean)**2
151  for x in sdata])) / len(sdata)
152  statistics['standard_deviation'].append((s**0.5))
153  return (statistics, series)

191  s = float(sum([(x-temp[0])**2

194  s = 'NA'

196  s = float(sum([(x-temp[0])**2
