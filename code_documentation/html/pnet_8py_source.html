26 class PNet(object):
27  '''!
28  Class to represent a Petri Net or Petri Net typed object.
29 
30  The places and transition rules are represented as dictionary
31  objects. Places dictionary will have the name of place as key and
32  the Place (pnet.Place object) as value. Transition rules
33  dictionary will have the name of rule appended with a number (in
34  the format of <rule name>_<number> in order to ensure uniqueness)
35  as key and the value is a dictionary to represent the transition
36  rule (the structure of the transition rule dictionary is dependent
37  on the type of rules).
38 
39  The following types of transition rules are allowed:
40  - step rule
41  - delay rule
42  - incubate rule
43  - ratio rule
44  - function rule
45 
46  Step rule is to be executed at each time step. For example, if 20g
47  of flour is to be transferred from flour bowl to mixer bowl at
48  each time step, this 'add_flour' rule can be defined as:
49 
50  >>> net.add_rules('add_flour', 'step', ['flour.flour -> mixer.flour; 20'])
51 
52  A single step rule can trigger more than one token movement. For
53  example, the following step rule simulates the mixing of
54  ingredients into a flour dough:
55 
56  >>> net.add_rules('blend', 'step', ['mixer.flour -> mixer.dough; 15', 'mixer.water -> mixer.dough; 10', 'mixer.sugar -> mixer.dough; 0.9', 'mixer.yeast -> mixer.dough; 1'])
57 
58  Delay rule acts as a time delay between each token movement. For
59  example, the following rule simulates the transfer of 0.5g of
60  yeast into the mixer bowl:
61 
62  >>> net.add_rules('add_yeast', 'delay', ['yeast.yeast -> mixer.yeast; 0.5; 10'])
63 
64  Incubate rule is a variation of delay rule. While delay rule is
65  not condition dependent, incubate rule starts a time delay when
66  one or more conditions are met. For example,
67 
68  >>> net.add_rules('rise', 'incubate', ['10; mixer.dough -> pan.dough; mixer.flour == 0; mixer.water == 0; mixer.sugar == 0; mixer.yeast == 0'])
69 
70  sets a 10 time step delay when all flour, water, sugar, and yeast
71  in the mixer bowl are used up, which simulates the completemixing
72  into a bread dough. The 10 time step then simulates the time
73  needed for the dough to rise. After 10 time steps, dough in the
74  mixer is transferred into the pan.
75 
76  Ratio rule is a variant of step rule. Instead of absolute number
77  of tokens to move, the movement is a percentage of the number of
78  tokens. For example,
79 
80  >>> net.add_rules('bake', 'ratio', ['pan.dough -> pan.bread; 0.3; pan.dough < 1; 0'])
81 
82  will move 30% of the token value from dough in pan to bread in
83  pan. If the token value of dough in pan is less than 1, then the
84  token value of dough in pan will be set to 0.
85 
86  Function rule is a generic and free-form rule, that takes the form
87  of a Python function. This is usually used when the transition
88  cannot be represented by any other rules. Given a user-defined
89  function, FUNC,
90 
91  >>> net.add_rules('cool', 'function', ['table.temperature -> air.temperature', FUNC, 'table.bread > 0; table.temperature > 30'])
92 
93  FUNC will be executed when table.bread > 0 and table.temperature >
94  30. The returned result of FUNC will be the token transfer from
95  table.temperature to air.temperature.
96 
97  FUNC takes all the places as a single parameter, such as FUNC(
98  places), where 'places' is a dictionary with the name of each
99  place as key. Token values in any place can be assessed. For
100  example,
101 
102  >>> place_names = places.keys()
103 
104  >>> a_place = places[places_names[0]]
105 
106  >>> token_set = a_place.attributes.keys()
107 
108  >>> a_token_value = a_place.attributes[token_set[0]]
109  '''

144  def add_rules(self, rule_name, rule_type, actions):
145  '''!
146  Method to add a transition rule into the Petri Net.
147 
148  @param rule_name string: name of the transition rule. This
149  name need not be unique within the model as this method will
150  append a running rule number to the name to ensure internal
151  uniqueness.
152  @param rule_type string: type of rule. Allowable types are
153  'step' for step rule, 'delay' for delay rule, and 'incubate'
154  for incubate rule. Please see module documentation for the
155  description of rules.
156  @param actions list: describe the action(s) of the transition
157  rule
158  '''
159  if rule_type not in ['function']:
160  for t in actions:
161  t = [x.strip() for x in t.split(';')]
162  d = {'type': rule_type,
163  'movement': None}
164  if rule_type == 'step':
165  movement = [x.strip() for x in t[0].split('->')]
166  d['movement'] = [(loc.split('.')[0], loc.split('.')[1])
167  for loc in movement]
168  d['value'] = float(t[1])
169  if rule_type == 'delay':
170  movement = [x.strip() for x in t[0].split('->')]
171  d['movement'] = [(loc.split('.')[0], loc.split('.')[1])
172  for loc in movement]
173  d['value'] = float(t[1])
174  d['delay'] = int(t[2])
175  if rule_type == 'incubate':
176  d['value'] = float(t[0])
177  movement = [x.strip() for x in t[1].split('->')]
178  d['movement'] = [(loc.split('.')[0], loc.split('.')[1])
179  for loc in movement]
180  d['conditions'] = [cond for cond in t[2:]]
181  d['timer'] = 0
182  if rule_type == 'ratio':
183  movement = [x.strip() for x in t[0].split('->')]
184  d['movement'] = [(loc.split('.')[0], loc.split('.')[1])
185  for loc in movement]
186  d['ratio'] = float(t[1])
187  d['limit_check'] = t[2]
188  d['limit_set'] = float(t[3])
189  self.rules[rule_name + '_' + str(self.rulenumber)] = d
190  self.rulenumber = self.rulenumber + 1
191  if rule_type in ['function']:
192  d = {'type': rule_type,
193  'movement': None}
194  if rule_type == 'function':
195  movement = [x.strip() for x in actions[0].split('->')]
196  d['movement'] = [(loc.split('.')[0], loc.split('.')[1])
197  for loc in movement]
198  d['function'] = actions[1]
199  d['conditions'] = [cond.strip()
200  for cond in actions[2].split(';')]
201  self.rules[rule_name + '_' + str(self.rulenumber)] = d
202  self.rulenumber = self.rulenumber + 1

204  def _step_rule(self, movement, value, interval):
205  '''!
206  Private method which simulates a step rule action.
207 
208  @param movement string: defines the movement of a token type.
209  Each movement is defined in the following format: <source
210  place>.<source token> -> <destination place>.<destination
211  token>
212  @param value float: the number of tokens to move
213  @param interval integer: simulation time interval
214  '''
215  source_place = self.places[movement[0][0]]
216  source_value = movement[0][1]
217  destination_place = self.places[movement[1][0]]
218  destination_value = movement[1][1]
219  if source_place.attributes[source_value] < (value*interval) \
220  and self.zerolowerbound == True:
221  value = source_place.attributes[source_value]
222  source_place.attributes[source_value] = \
223  source_place.attributes[source_value] - (value*interval)
224  destination_place.attributes[destination_value] = \
225  destination_place.attributes[destination_value] + \
226  (value*interval)

228  def _test_condition(self, place, token, operator, value):
229  '''!
230  Private method used by rule processors for logical check of
231  condition. For example, the condition 'mixer.flour == 0' will
232  be written as
233 
234  >>> _test_condition('mixer', 'flour', '==', 0)
235 
236  @param place string: name of place/container
237  @param token string: name of token
238  @param operator string: binary operator. Allowable values are
239  '==' (equals to), '>' (more than), '>=' (more than or equals
240  to), '<' (less than), '<=' (less than or equals to), and '!=' (
241  not equals to).
242  @param value: value to be checked
243  @return 'passed' if test result is true, or 0 if test result
244  is false
245  '''
246  value = float(value)
247  if operator == '==' and \
248  self.places[place].attributes[token] == value:
249  return 'passed'
250  elif operator == '>' and \
251  self.places[place].attributes[token] > value:
252  return 'passed'
253  elif operator == '>=' and \
254  self.places[place].attributes[token] >= value:
255  return 'passed'
256  elif operator == '<' and \
257  self.places[place].attributes[token] < value:
258  return 'passed'
259  elif operator == '<=' and \
260  self.places[place].attributes[token] <= value:
261  return 'passed'
262  elif operator == '!=' and \
263  self.places[place].attributes[token] != value:
264  return 'passed'
265  else:
266  return 'failed'

268  def _conditions_processor(self, conditions):
269  '''!
270  Private method used by rule processors to evaluate logical
271  conditions.
272 
273  @param conditions list: one or more logical conditions in the
274  format of '<place>.<token> <binary operator> <criterion>',
275  such as 'oven.heat > 300', for evaluation
276  '''
277  test = [0] * len(conditions)
278  for i in range(len(conditions)):
279  cond = conditions[i]
280  if len(cond.split('==')) == 2:
281  operator = '=='
282  cond = [c.strip() for c in cond.split('==')]
283  elif len(cond.split('>')) == 2:
284  operator = '>'
285  cond = [c.strip() for c in cond.split('>')]
286  elif len(cond.split('>=')) == 2:
287  operator = '>='
288  cond = [c.strip() for c in cond.split('>=')]
289  elif len(cond.split('<')) == 2:
290  operator = '<'
291  cond = [c.strip() for c in cond.split('<')]
292  elif len(cond.split('<=')) == 2:
293  operator = '<='
294  cond = [c.strip() for c in cond.split('<=')]
295  elif len(cond.split('!=')) == 2:
296  operator = '!='
297  cond = [c.strip() for c in cond.split('!=')]
298  source_place = cond[0].split('.')[0]
299  source_value = cond[0].split('.')[1]
300  criterion = cond[1]
301  test[i] = self._test_condition(source_place, source_value,
302  operator, criterion)
303  return test

305  def _incubate_rule(self, rule, interval):
306  '''!
307  Private method which simulates an incubate rule action.
308 
309  @param rule: a dictionary representing the incubate rule
310  @param interval integer: simulation time interval
311  @return modified rule dictionary
312  '''
313  value = rule['value']
314  timer = rule['timer']
315  conditions = rule['conditions']
316  movement = rule['movement']
317  test = self._conditions_processor(conditions)
318  if len(['failed' for t in test if t == 'failed']) == 0:
319  if (timer + interval) < value:
320  rule['timer'] = timer + interval
321  else:
322  source_place = self.places[movement[0][0]]
323  source_value = movement[0][1]
324  destination_place = self.places[movement[1][0]]
325  destination_value = movement[1][1]
326  destination_place.attributes[destination_value] = \
327  destination_place.attributes[destination_value] + \
328  source_place.attributes[source_value]
329  source_place.attributes[source_value] = 0
330  rule['timer'] = 0
331  return rule

333  def _ratio_rule(self, movement, ratio, limit_check,
334  limit_set, interval):
335  '''!
336  Private method which simulates a ratio rule action.
337 
338  @param movement string: defines the movement of a token type.
339  Each movement is defined in the following format: <source
340  place>.<source token> -> <destination place>.<destination
341  token>
342  @param ratio float: the ratio of tokens to move
343  @param limit_check string: logical check for remainder value
344  @param limit_set flaot: value to set token if token in
345  limit_check is true
346  @param interval integer: simulation time interval
347  '''
348  source_place = self.places[movement[0][0]]
349  source_value = movement[0][1]
350  destination_place = self.places[movement[1][0]]
351  destination_value = movement[1][1]
352  # Step 1: Perform ratio rule operation
353  token_value = source_place.attributes[source_value] * \
354  ratio * interval
355  source_place.attributes[source_value] = \
356  source_place.attributes[source_value] - token_value
357  destination_place.attributes[destination_value] = \
358  destination_place.attributes[destination_value] + token_value
359  # Step 2: Perform remaining checks and corrections
360  if len(limit_check.split('>')) == 2:
361  operator = '>'
362  limit_check = [c.strip() for c in limit_check.split('>')]
363  if len(limit_check.split('<')) == 2:
364  operator = '<'
365  limit_check = [c.strip() for c in limit_check.split('<')]
366  check_place = limit_check[0].split('.')[0]
367  check_token = limit_check[0].split('.')[1]
368  check_value = float(limit_check[1])
369  if self._test_condition(check_place, check_token,
370  operator, check_value) == 'passed':
371  if source_value in self.losses:
372  self.losses[source_value] = self.losses[source_value] + \
373  source_place.attributes[source_value] - limit_set
374  else:
375  self.losses[source_value] = \
376  source_place.attributes[source_value] - limit_set
377  source_place.attributes[source_value] = limit_set

406  def _execute_rules(self, clock, interval):
407  '''!
408  Private method used by PNet.simulate() and
409  PNet.simulate_yield() to execute all the rules.
410 
411  @param clock cloat: wall time of the current simulation
412  @param interval integer: simulation time interval
413  '''
414  affected_places = []
415  for rName in self.rules.keys():
416  # Step rule
417  if self.rules[rName]['type'] == 'step':
418  movement = self.rules[rName]['movement']
419  value = self.rules[rName]['value']
420  self._step_rule(movement, value, interval)
421  # Delay rule
422  if self.rules[rName]['type'] == 'delay' and \
423  (clock % self.rules[rName]['delay']) == 0:
424  movement = self.rules[rName]['movement']
425  value = self.rules[rName]['value']
426  self._step_rule(movement, value, interval)
427  # Incubate rule
428  if self.rules[rName]['type'] == 'incubate':
429  value = self.rules[rName]['value']
430  rule = self._incubate_rule(self.rules[rName], interval)
431  self.rules[rName] = rule
432  # Ratio rule
433  if self.rules[rName]['type'] == 'ratio':
434  movement = self.rules[rName]['movement']
435  ratio = self.rules[rName]['ratio']
436  limit_check = self.rules[rName]['limit_check']
437  limit_set = self.rules[rName]['limit_set']
438  self._ratio_rule(movement, ratio, limit_check,
439  limit_set, interval)
440  # Function rule
441  if self.rules[rName]['type'] == 'function':
442  movement = self.rules[rName]['movement']
443  conditions = self.rules[rName]['conditions']
444  function = self.rules[rName]['function']
445  self._function_rule(movement, function, conditions)

447  def simulate(self, end_time, interval=1.0, report_frequency=1.0):
448  '''!
449  Method to simulate the Petri Net. This method stores the
450  generated report in memory; hence, not suitable for extended
451  simulations as it can run out of memory. It is possible to
452  conserve memory by reducing the reporting frequency. Use
453  simulate_yield method for extended simulations.
454 
455  @param end_time integer: number of time steps to simulate. If
456  end_time = 1000, it can be 1000 seconds or 1000 days,
457  depending on the significance of each step
458  @param interval float: number of intervals between each time
459  step. Default = 1.0, simulate by time step interval
460  @param report_frequency float: number of time steps between
461  each reporting. Default = 1.0, each time step is reported
462  '''
463  clock = 1
464  end_time = int(end_time)
465  while clock < (end_time + 1):
466  self._execute_rules(clock, interval)
467  if (clock % report_frequency) == 0:
468  self._generate_report(clock)
469  clock = clock + interval

471  def simulate_yield(self, end_time, interval=1.0):
472  '''!
473  Method to simulate the Petri Net. This method runs as a
474  generator, making it suitable for extended simulation.
475 
476  @param end_time integer: number of time steps to simulate. If
477  end_time = 1000, it can be 1000 seconds or 1000 days,
478  depending on the significance of each step
479  @param interval float: number of intervals between each time
480  step. Default = 1.0, simulate by time step interval
481  '''
482  clock = 1
483  end_time = int(end_time)
484  while clock < end_time:
485  self._execute_rules(clock, interval)
486  self._generate_report(clock)
487  rept = {}
488  for k in self.report[str(clock)].keys():
489  rept[k] = self.report[str(clock)][k]
490  del self.report[str(clock)][k]
491  yield (clock, rept)
492  clock = clock + interval

494  def _generate_report(self, clock):
495  '''!
496  Private method to generate and store report in memory of each
497  token status (the value of each token) in every place/
498  container.
499

501  '''
502  rept = {}
503  for pName in self.places.keys():
504  for aName in self.places[pName].attributes.keys():
505  value = self.places[pName].attributes[aName]
506  name = '.'.join([pName, aName])
507  rept[name] = value
508  self.report[str(clock)] = rept

510  def report_tokens(self, reportdict=None):
511  '''!
512  Method to report the status of each token(s) from each place
513  as a list. This can be used in 2 different ways: to generate a
514  list representation of a status from one time step (such as
515  from simulate_yield method), or to generate a list
516  representation of a status from entire simulation (such as
517  from simulate method).
518 
519  from simulate method
520  >>> net.simulate(65, 1, 1)
521 
522  >>> status = net.report_tokens()
523 
524  from simulate_yield method
525  >>> status = [d for d in net.simulate_yield(65, 1)]
526 
527  >>> status = [(d[0], net.report_tokens(d[1])) for d in status]
528 
529  @param reportdict dictionary: status from one time step.
530  Default = None. If None, it will assume that simulate method
531  had been executed and all status are stored in memory, and
532  this method will generate a report from status stored in memory
533  @return tuple of ([<place.token name>], [([<place.token
534  value>]]) if reportdict is given, or tuple of (time step, [<
535  place.token name>], [([<place.token value>]]) if reportdict is
536  None.
537  '''
538  if reportdict:
539  placetokens = reportdict.keys()
540  tokenvalues = [reportdict[k] for k in placetokens]
541  return (placetokens, tokenvalues)
542  else:
543  timelist = list(self.report.keys())
544  datalist = [0] * len(timelist)
545  for i in range(len(timelist)):
546  placetokens = list(self.report[timelist[i]].keys())
547  tokenvalues = [self.report[timelist[i]][k]
548  for k in placetokens]
549  datalist[i] = (timelist[i], placetokens, tokenvalues)
550  return datalist
