10 class OptimizationTarget(object):
11  '''
12  Abstract class to create the object to be optimized. This class
13  is used to created an inherited class, which is the optimization
14  target. In the context of genetic algorithms, the inherited class
15  represents an organism where its chromosome(s) is/are being
16  optimized - so that the reduced execution results (comparatorData)
17  approaches targetResults.
18 
19  The way the organism executes is as follows: Firstly, The
20  runnerFunction in the inherited class represents the method to
21  execute the chromosomes to generate the executionResults. Secondly,
22  the dataFunction in the inherited class represents the method to
23  select required data variables (reduce the number of data variables)
24  from executionResults into comparatorData. Finally, the
25  comparatorFunction in the inherited class compares between the
26  targetResults and comparatorData to generate a fitnessScore. The higher
27  the fitness score, the fitter the organism and the higher chances it
28  gets into the next generation.
29 
30  There can be one or more chromosomes in each organism; hence,
31  chromosomes are defined as dictionary in this template class.
32 
33  The chromosomes_lower_bounds and chromosomes_upper_bounds
34  represents the lower and upper bound values of the chromosomes,
35  which can be used during mutation.
36  '''

89 class OptimizerGA(object):
90  '''
91  Genetic algorithm (GA) optimizer class.
92 
93  This class takes an OptimizationTarget object (which represents an
94  individual organism) and replicates it to the required population size
95  and store them in population dictionary (where the keys represents the
96  sample IDs). These sample IDs will be reused over generations; hence,
97  have to take account of population count to be unique. Two functions
98  needs to be defined - mutate (which mutates the chromosomes in
99  population[ID].chromosomes and are bounded by population[ID].
100  chromosomes_lower_bounds and population[ID].chromosomes_upper_bounds
101  for lower and upper value boundaries respectively), and mate (which
102  eliminates unfit organisms and mates fit organisms for the next
103  generation). There is a defined set of mating and mutation methods
104  pre-defined but these can be overrode. The run method performs GA
105  optimization on the population.
106  '''

117  (Default = 10).
118  @type max_generations: integer

174  def run(self):
175  '''
176  Method to run the GA optimizer, which will execute the following steps
177  until one of the organism is fitted (OptimizationTarget.fitted == True)
178  or the maximum generation is reached:
179  - for each organism, execute runnerFunction()
180  - for each organism, execute dataFunction()
181  - for each organism, execute comparatorFunction()
182  - if none of the organisms is fitted, execute mutate function and
183  mate function
184 
185  However, all organisms in the population will undergo mutation before
186  the first execution. This is to prevent all organisms from getting the
187  same fitness score at the first generation as all organisms are clones
188  of each other at instantiation.
189 
190  @return: (generation count, population dictionary)
191  '''
192  def runReport(population):
193  popFitness = {}
194  for k in list(population.keys()):
195  popFitness[k] = population[k].fitnessScore
196  averageFitness = [population[k].fitnessScore
197  for k in list(population.keys())]
198  bestFitness = max(averageFitness)
199  averageFitness = sum(averageFitness) / float(len(averageFitness))
200  print('Generation %s, Average Fitness: %.7f, Best Fitness: %.7f' % \
201  (self.generations, averageFitness, bestFitness))
202  if int(self.verbose) > 2:
203  print('Organism Fitness Scores: ' + str(popFitness))
204  if self.generations == 0:
205  self.population = self.mutateFunction(self.population)
206  while (self.generations < self.max_generations):
207  for i in range(len(self.population)):
208  self.population[i].runnerFunction()
209  self.population[i].dataFunction()
210  self.population[i].comparatorFunction()
211  runReport(self.population)
212  self._saveExtremes()
213  if True in [self.population[i].fitted
214  for i in range(len(self.population))]:
215  return (self.generation, self.population)
216  self.population = self.mutateFunction(self.population)
217  self.population = self.mateFunction(self.population)
218  for i in range(len(self.population)):
219  self.population[i].modifierFunction()
220  self.generations = self.generations + 1

242  def _mutateRandom(self, population):
243  '''
244  Private method - random mutation scheme (mutation scheme = random). In
245  this scheme, each of the chromosome is randomly mutated. A random float
246  (multiplier) between 0 and 2 will be generated for each element on the
247  chromosome. If the multiplier is lower than 1, then the data value of
248  the element on the chromosome will be reduced. if the multipler is more
249  than 1, then the data value of the element on the chromosome will be
250  increased.
251 
252  @param population: population to mutate
253  @type population: dictionary
254  @return: mutated population
255  '''
256  def mutateChromosome(chr, lower, upper):
257  position = range(len(chr))
258  position = [random.choice(position)
259  for i in range(int(len(chr)*self.mutationRate))]
260  for i in position:
261  multiplier = random.randint(0, 2000) / float(1000)
262  if multiplier < 1:
263  gap = chr[i] - lower[i]
264  chr[i] = chr[i] - (gap * multiplier)
265  else:
266  multiplier = multiplier - 1
267  gap = upper[i] - chr[i]
268  chr[i] = chr[i] + (gap * multiplier)
269  return chr
270  def mutate(organism):
271  for k in organism.chromosomes.keys():
272  chr = organism.chromosomes[k]
273  lower = organism.chromosomes_lower_bounds[k]
274  upper = organism.chromosomes_upper_bounds[k]
275  organism.chromosomes[k] = mutateChromosome(chr, lower, upper)
276  return organism
277  for k in population.keys():
278  population[k] = mutate(population[k])
279  return population

295  Private method - top 50% Fission mating scheme (mating scheme =
296  top50fission). In this scheme, organisms with fitness score lower than
297  the average fitness score of the population will be removed. The
298  remaining organisms will be randomly selected for cloning / duplication
299  to fill up the population.
300 
301  @param population: population to mate
302  @type population: dictionary
303  @return: mated population
304  '''
305  def kill(population):
306  meanfitness = [population[k].fitnessScore
307  for k in self.population.keys()]
308  meanfitness = sum(meanfitness) / len(meanfitness)
309  setKill = []
310  for k in self.population.keys():
311  if population[k].fitnessScore < meanfitness:
312  setKill.append(k)
313  for k in setKill:
314  del population[k]
315  newpop = {}
316  count = 0
317  for k in self.population.keys():
318  newpop[count] = population[k]
319  count = count + 1
320  return newpop
