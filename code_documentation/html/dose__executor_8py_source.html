17 def set_instruction_version(ragaraja_version,
18  instruction_set='ragaraja_instructions.txt'):
19  '''
20  Set active / usable set of Ragaraja instructions / operations for
21  the current simulation.
22 
23  @param ragaraja_version: defines the version of Ragaraja instructions
24  to be used. Allowed values are {0 | 0.1 | 1}
25  @param instruction_set: file name for user-defined instruction set
26  usage. This will only be used when ragaraja_version=0. Default =
27  ragaraja_instructions.txt. Please see ragaraja_instructions.txt for
28  format.
29  @type instruction_set: string
30  @return: dictionary of ragaraja instructions as keys and instruction
31  execution functions as values.
32 
33  @since: version 0.4.1
34  '''
35  if ragaraja_version == 0:
36  f = open(instruction_set, 'r').readlines()
37  f = [x[:-1].split('=') for x in f]
38  f = [x[0] for x in f if x[1] == 'Y']
39  ragaraja_instructions = f
40  if ragaraja_version == 0.1:
41  ragaraja_instructions = N.nBF_instructions
42  if ragaraja_version == 1:
43  ragaraja_instructions = N.ragaraja_v1
44  for instruction in N.ragaraja:
45  if instruction not in ragaraja_instructions:
46  N.ragaraja[instruction] = N.not_used
47  return ragaraja_instructions
