514 class Matrix(object):
515  '''
516  A Matrix class that is implemented as a sparse matrix.
517 
518  Mathematically speaking, there is no difference between a dense matrix
519  (most values are non-zero) and a sparse matrix (most values are zero).
520  However, there are implementation differences and memory usage
521  differences between sparse and dense matrix. In this case, this
522  implementation considers all matrix as sparse and stores the values in
523  a coordinate list.
524 
525  This class is based on the work of Bill McNeill (http://
526  aspn.activestate.com/ASPN/Cookbook/Python/Recipe/189971) and Alexander
527  Pletzer (http://aspn.activestate.com/ASPN/Cookbook/Python/Recipe/52275).
528  '''
529 
530  def __init__(self, *args):
531  '''
532  Constructor method. Bear in mind that Python uses zero-index; that
533  is, the first element of the first row is has the coordinate of
534  (0,0) rather than (1,1) in regular matrix notations.
535 
536  There are 4 ways to construct a matrix.
537 
538  Firstly, a null matrix (no data elements) can be constructed as
539 
540  >>> m = Matrix()
541 
542  Secondly, a zero matrix can be constructed as
543 
544  >>> m = Matrix(2)
545 
546  This will result in a zero square matrix, [[0, 0], [0, 0]]
547 
548  A non-square matrix can be constructed by providing the (row,
549  column) pairs. For example,
550 
551  >>> m = Matrix(2, 3)
552 
553  will result in [[0, 0, 0], [0, 0, 0]]
554 
555  Thirdly, a matrix can be constructed by providing a list of list in
556  the format of [[first row], [second row], ... [last row]]
557 
558  >>> m = Matrix([[1, 2, 3], [4, 5, 6]])
559 
560  Lastly, a matrix can be provided by providing a dictionary
561  non-zero values using (row, column) as key,
562 
563  >>> m = Matrix({(0,0): 3,
564  ... (1,1): 6})
565 
566  will result in [[3, 0], [0, 6]].
567 
568  @param args: arguments for matrix construction. Please see above
569  for description.
570  '''
571  self.values = {}
572  self.dimensions = [0, 0]
573  if len(args) == 0:
574  pass
575  if len(args) == 1 and isinstance(args[0], types.IntType):
576  self.createNullMatrix(args[0], args[0])
577  if len(args) == 2 and isinstance(args[0], types.IntType) and \
578  isinstance(args[1], types.IntType):
579  self.createNullMatrix(args[0], args[1])
580  if len(args) == 1 and isinstance(args[0], types.ListType):
581  for row in range(len(args[0])):
582  if isinstance(args[0][row], types.ListType):
583  self.addReplaceRow(row, args[0][row])
584  self.updateDimensions()
585  if len(args) == 1 and isinstance(args[0], types.DictType):
586  for k in args[0].keys():
587  self.addReplaceElement(k, args[0][k])
588  self.updateDimensions()
589 
590  def updateDimensions(self):
591  '''
592  Method to update the dimension of the matrix.
593  '''
594  row_count = set()
595  column_count = set()
596  for coordinate in self.values.keys():
597  row_count.update([coordinate[0]])
598  column_count.update([coordinate[1]])
599  self.dimensions[0] = max(list(row_count)) + 1
600  self.dimensions[1] = max(list(column_count)) + 1
601 
602  def createNullMatrix(self, rows, columns):
603  '''
604  Method to create a null (zero) matrix.
605 
606  @param rows: number of rows.
607  @type rows: integer
608  @param columns: number of columns.
609  @type columns: integer
610  '''
611  rows = int(rows)
612  columns = int(columns)
613  for r in range(rows):
614  for c in range(columns):
615  self.values[(r, c)] = 0
616  self.dimensions = [rows, columns]
617 
618  def createIdentityMatrix(self, size):
619  '''
620  Method to create an identity matrix.
621 
622  @param size: size (number of rows and columns) of identity matrix.
623  @type size: integer
624  '''
625  size = int(size)
626  for r in range(size):
627  for c in range(size):
628  if r == c:
629  self.values[(r, c)] = 1.0
630  else:
631  self.values[(r, c)] = 0.0
632  self.dimensions = [size, size]

634  def addReplaceRow(self, row_number, row_data, update_dimensions=False):
635  '''
636  Method to add a row (if the row does not exist) or to replace a row
637  (if the row exist) in the matrix.
638 
639  @param row_number: the row number to add or replace, based on
640  zero-index (the first row is zero-th row).
641  @type row_number: integer
642  @param row_data: data for the row to add or replace.
643  @type row_data: list
644  @param update_dimensions: flag to determine whether to update
645  matrix dimensions, which can reduce speed if matrix is large.
646  Default = false (dimensions not updated).
647  @type update_dimensions: boolean
648  '''
649  row_number = int(row_number)
650  for c in range(len(row_data)):
651  self.values[(row_number, c)] = row_data[c]
652  if update_dimensions:
653  self.updateDimensions()

655  def addReplaceElement(self, coordinate, value, update_dimensions=False):
656  '''
657  Method to add an element (if the element does not exist) or to
658  replace the element (if the element exist) in the matrix.
659 
660  @param coordinate: (row, column) coordinate to add or replace,
661  based on zero-th index (first element of first row is zero-th row
662  zero-th column).
663  @type coordinate: list or tuple
664  @param value: value of the coordinate to add or replace.
665  @param update_dimensions: flag to determine whether to update
666  matrix dimensions, which can reduce speed if matrix is large.
667  Default = false (dimensions not updated).
668  @type update_dimensions: boolean
669  '''
670  if len(coordinate) == 2:
671  r = int(coordinate[0])
672  c = int(coordinate[1])
673  self.values[(r, c)] = value
674  if update_dimensions:
675  self.updateDimensions()

773  '''
