24 def binarize(data, absent=0, type='Full'):
25  """
26  Converts input data in a list of presence or absence of values.
27  For example,
28  binarize([1, 2, 0, 3, 4, 0], 0, 'Full') --> [1, 1, 0, 1, 1, 0]
29  binarize([1, 2, 0, 3, 4, 0], 0, 'Partial') --> [1, 2, 0, 3, 4, 0]
30  binarize([1, 2, 0, 3, 4, 0], 2, 'Full') --> [1, 0, 1, 1, 1, 1]
31  binarize([1, 2, 0, 3, 4, 0], 2, 'Partial') --> [1, 0, 0, 3, 4, 0]
32 
33  As 1 and 0 are commonly used to denote presence and absence, please
34  take care when data contains 1s and 0s.
35 
36  @param data: data to binarize
37  @type data: list
38  @param absent: value/symbol to denote absent value. Default = 0.
39  @param type: Denotes type of binarization, which can be 'Full' or
40  'Partial'. If full binarize, the returned data will only be 1
41  (denote presence) or 0 (denote absence). If partial binarize, the
42  returned data will be 0 (denote absence) or the original values
43  in data (denoting not absent). Default = Full.
44  """
45  if type == 'Full':
46  return [{absent: 0}.get(x, 1) for x in data]
47  elif type == 'Partial':
48  return [{absent: 0}.get(x, x) for x in data]

50 def compare(original, test, absent, type='Set'):
51  """
52  Used for processing list-based (positional) or set-based
53  (non-positional) distance of categorical data.
54 
55  @param original: list of original data
56  @param test: list of data to test against original
57  @param absent: indicator to define absent data
58  @param type: {List | Set}
59  """
60  if type == 'Set':
61  original_only = float(len([x for x in original if x not in test]))
62  test_only = float(len([x for x in test if x not in original]))
63  both = float(len([x for x in original if x in test]))
64  return (original_only, test_only, both, 0.0)
65  if type == 'List':
66  original, test = list(original), list(test)
67  original_only, test_only, both, none = 0.0, 0.0, 0.0, 0.0
68  for i in range(len(original)):
69  if original[i] == absent and test[i] == absent:
70  none = none + 1
71  elif original[i] == test[i]:
72  both = both + 1
73  elif original[i] != absent and test[i] == absent:
74  original_only = original_only + 1
75  elif original[i] == absent and test[i] != absent:
76  test_only = test_only + 1
77  else: pass
78  return (original_only, test_only, both, none)

595 def Stiles(original, test, absent=0, type='Set'):
596  """
597  Stiles coefficient for nominal or ordinal data.
598 
599  Coefficient:
600  M{log10((P(|(A * D) - (B * C)| - (P / 2)) ^ 2) /
601  (A + B)(A + C)(B + D)(C + D))}
602 
603  @param original: list of original data
604  @param test: list of data to test against original
605  @param absent: user-defined identifier for absent of region,
606  default = 0
607  @param type: {Set | List}, define whether use Set comparison
608  (non-positional) or list comparison (positional), default = Set
609 
610  @see: Ling, MHT. 2010. COPADS, I: Distances Measures between Two
611  Lists or Sets. The Python Papers Source Codes 2:2.
612 
613  @status: Tested function
614  @since: version 0.3
615  """
616  (original, test, both, none) = compare(original, test, absent, type)
617  n = original + test + both + none
618  t = abs((both * none) - (test * original))

701 def Tulloss(original, test, absent=0, type='Set'):
702  """
703  Tulloss coefficient for nominal or ordinal data.
704 
705  Coefficient:
706  M{sqrt(U * S * R)}
707  - M{U = log(1 + ((min(B, C) + A) / (max(B, C) + A))) / log2}
708  - M{S = 1 / sqrt(log(2 + (min(B, C) / (A + 1))) / log2)}
709  - M{R = log(1 + A / (A + B))log(1 + A / (A + C)) / log2log2}
710 
711  @param original: list of original data
712  @param test: list of data to test against original
713  @param absent: user-defined identifier for absent of region,
714  default = 0
715  @param type: {Set | List}, define whether use Set comparison
716  (non-positional) or list comparison (positional), default = Set
717 
718  @see: Ling, MHT. 2010. COPADS, I: Distances Measures between Two
719  Lists or Sets. The Python Papers Source Codes 2:2.
720 
721  @status: Tested function
722  @since: version 0.3
723  """
724  (original, test, both, none) = compare(original, test, absent, type)
725  U = (min(test, original) + both) / (max(test, original) + both)
726  U = math.log10(1 + U) / math.log10(2)
727  S = math.log10(2 + (min(test, original) / (both + 1)))
728  S = 1 / ((S / math.log10(2)) ** 0.5)
729  R = math.log10(1 + (both / (both + original)))
730  R = R * math.log10(1 + (both / (both + test)))
731  R = R / (math.log10(2) * math.log10(2))
732  return (U * S * R) ** 0.5
