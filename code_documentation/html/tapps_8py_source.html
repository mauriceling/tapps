73 def LoadPlugin(session, plugin_name):
74  '''!
75  Function to perform basic checks and load a plugin into session
76  dictionary to get it ready for use.
77 
78  The following checks are done:
79  1. Presence of callable main function in plugin
80  2. Presence of Parameters dictionary
81  3. Presence of instructions
82  4. Presence of plugin name
83  5. Presence of release number
84  6. Presence of acceptable category
85  7. Presence of short description
86  8. Presence of long description
87  9. Presence of URL
88  10. Presence of contact details
89  11. Presence of license
90 
91  The following changes will be made to session dictionary:
92  1. If plugin is successfully loaded, the plugin name will be
93  appended to session['plugins']['loaded']
94  2. If plugin is successfully loaded, the plugin name will be
95  appended to session['plugins'][<plugin category>]
96  3. If plugin is successfully loaded, details of the plugin
97  will be loaded into session['plugins_<plugin name>'], and
98  session['plugins_<plugin name>']['main'] will contain the
99  function entry point for the plugin
100  4. If plugin is NOT successfully loaded (failure in one or
101  more of the above checklist), the checklist for the plugin
102  will be loaded into session['plugins']['loadFail'][<plugin
103  name>]
104 
105  @param session: dictionary to hold all data within the current
106  session. Please see module documentation for more details.
107  @param plugin String: module name of plugin to load (corresponding
108  to the folder/dictionary which the plugin resides - <current
109  working directory>/plugin/<plugin folder name>)
110  @return: session dictionary
111  '''
112  session = e.LoadPlugin(session, plugin_name)
113  return session

115 def RunPlugin(session, parameter_name):
116  '''!
117  Function to run/execute a plugin using the parameters (
118  sub-dictionary within session, under 'parameters' key) needed for
119  the plugin and returning the execution results into session
120  dictionary.
121 
122  The parameters sub-dictionary will need to contain values for the
123  following keys:
124  - analysis_name: user-given name for the analytical execution
125  of plugin
126  - plugin_name: name of plugin to execute
127  - dataframe: a dataframe object to act as data for use by the
128  plugin and any other plugin-specific parameters/options.
129 
130  After execution, analysis results will be returned as value to
131  'results' key in the parameters sub-dictionary.
132 
133  Hence, session['parameters'][<parameter_name>] will hold the
134  entire parameters sub-dictionary for the current plugin execution
135  and session['parameters'][<parameter_name>]['results'] will hold
136  the plugin output results.
137 
138  @param session: dictionary to hold all data within the current
139  session. Please see module documentation for more details.
140  @param parameter_name: name of parameter dictionary to hold all
141  the parameters needed to execute the plugin. Please see module
142  documentation for more details.
143  @return: session dictionary
144  '''
145  session = e.RunPlugin(session, parameter_name)
146  return session

148 def LoadCSV(session, filepath, series_header=True, separator=',',
149  fill_in=None, newline='\n'):
150  '''!
151  Function to load a comma-delimited (CSV) file as a data frame.
152 
153  The data frame will have the filepath as name, and will be loaded
154  into session dictionary under 'new_dataframe' key.
155 
156  @param session: dictionary to hold all data within the current
157  session. Please see module documentation for more details.
158  @param filepath String: path to CSV file.
159  @param series_header: boolean flag to denote whether the first row
160  in the CSV file contains the data header. It is highly recommended
161  that header is included in the CSV file.
162  @param separator: item separator within the CSV file.
163  @param fill_in: value to fill into missing values during process.
164  This is required as the number of data elements across each label
165  must be the same. Hence, filling in of missing values can occur
166  when (1) the newly added data series consists of new labels which
167  are not found in the current data frame (this will require filling
168  in of missing values in the current data frame), or (2) the
169  current data frame consists of labels that are not found in the
170  newly added data series (this will require filling in of missing
171  values to the newly added data series).
172  @param newline: character to denote new line or line feed in the
173  CSV file.
174  '''
175  session = e.LoadCSV(session, filepath, series_header, separator,
176  fill_in, newline)
177  return session

214 def RunScript(session, scriptfile):
215  '''!
216  Function to execute script file written in TAPPS language.
217 
218  @param session: dictionary to hold all data within the current
219  session. Please see module documentation for more details.
220  @param scriptfile: absolute path to TAPPS script file in the
221  format of a list.
222  '''
223  dirname = scriptfile[:-1]
224  print('')
225  print('Executing script file: %s' % os.sep.join(scriptfile))
226  print('')
227  def process_script(scriptfile):
228  scriptfile = os.sep.join(scriptfile)
229  print('')
230  print('Reading script file: %s' % scriptfile)
231  print('')
232  script = open(scriptfile, 'r').readlines()
233  script = [x[:-1] for x in script]
234  script = [x.strip() for x in script]
235  for i in range(len(script)):
236  if script[i].startswith('@include'):
237  f = script[i].split(' ')[1].strip()
238  script[i] = process_script(dirname + [f])
239  return script
240  def flatten(container):
241  for i in container:
242  if isinstance(i, list) or isinstance(i, tuple):
243  for j in flatten(i):
244  yield j
245  else:
246  yield i
247  fullscript = process_script(scriptfile)
248  fullscript = list(flatten(fullscript))
249  fullscript = [x for x in fullscript if x != '']
250  fullscript = [x for x in fullscript if not x.startswith('#')]
251  shell = s.Shell(session, 'TAPPS')
252  shell.cmdscript(fullscript)
