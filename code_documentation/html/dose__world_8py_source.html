12 class World(object):
13  '''
14  Representation of a 3-dimensional ecological world.
15 
16  The ecosystem is made up of ecological cells. Each ecological cell is
17  modelled as a dictionary of
18  - local_input: A list containing processed input, representing
19  the partial local ecological condition, to be used as input to
20  the organisms in the current ecological cell. This is updated
21  by World.update_local function.
22  - local_output: A list containing processed output, representing
23  the partial local ecological condition. This is updated by
24  World.update_local function.
25  - temporary_input: A list acting as temporary holding for input
26  after being fed to the organisms in the current ecological
27  cell, which is to be used to update local_input and local_output
28  lists by World.update_local and World.update_ecology functions.
29  - temporary_output: A list acting as temporary holding for output
30  from the organisms in the current ecological cell, which is to
31  be used to update local_input and local_output lists by
32  World.update_local and World.update_ecology functions.
33  - organisms: The number of organisms in the current ecological
34  cell which is updated by World.organism_movement and
35  World.organism_location functions.
36 
37  @see: Ling, MHT. 2012. An Artificial Life Simulation Library Based on
38  Genetic Algorithm, 3-Character Genetic Code and Biological Hierarchy.
39  The Python Papers 7: 5.
40  '''

43  def __init__(self, world_x, world_y, world_z):
44  '''
45  Setting up the world and ecosystem
46 
47  @param world_x: number of ecological cells on the x-axis
48  @type world_x: integer
49  @param world_y: number of ecological cells on the y-axis
50  @type world_y: integer
51  @param world_z: number of ecological cells on the z-axis
52  @type world_z: integer
53  '''
54  eco_cell = {'local_input': [], 'local_output': [],
55  'temporary_input': [], 'temporary_output': [],
56  'organisms': 0}
57  self.world_x = int(world_x)
58  self.world_y = int(world_y)
59  self.world_z = int(world_z)
60  for x in range(self.world_x):
61  eco_x = {}
62  for y in range(self.world_y):
63  eco_y = {}
64  for z in range(self.world_z):
65  eco_y[z] = copy.deepcopy(eco_cell)
66  eco_x[y] = copy.deepcopy(eco_y)
67  self.ecosystem[x] = copy.deepcopy(eco_x)
