72  def formatExceptionInfo(self, maxTBlevel=10):
73  '''!
74  Method to gather information about an exception raised. It is
75  used to readout the exception messages and type of exception.
76  This method takes a parameter, maxTBlevel, which is set to 10,
77  which defines the maximum level of tracebacks to recall.
78 
79  This method is obtained from http://www.linuxjournal.com/
80  article.php?sid=5821
81  '''
82  cla, exc, trbk = sys.exc_info()
83  excName = cla.__name__
84  try: excArgs = exc.__dict__["args"]
85  except KeyError: excArgs = "<no args>"
86  excTb = traceback.format_tb(trbk, maxTBlevel)
87  return (excName, excArgs, excTb)

154  def do_set(self, operand):
155  '''!
156  Method to set options in environment or within plugin
157  parameters set.
158 
159  Operand = [<object to show>]
160 
161  @param operand list: bytecode operand(s), if any
162  '''
163  op = operand[0].lower()
164  if op == 'displayast':
165  if operand[1].lower() in ['t', 'true']:
166  self.environment['display_ast'] = True
167  if operand[1].lower() in ['f', 'false']:
168  self.environment['display_ast'] = False
169  if op == 'cwd':
170  self.environment['cwd'] = operand[1]
171  self.session['cwd'] = operand[1]
172  if op == 'ocwd':
173  self.environment['cwd'] = self.environment['ocwd']
174  self.session['cwd'] = self.environment['ocwd']
175  if op == 'rcwd':
176  cwd = os.sep.join([self.environment['cwd'], operand[1]])
177  self.environment['cwd'] = cwd
178  self.session['cwd'] = cwd
179  if op == 'separator':
180  self.environment['separator'] = operand[1]
181  if op == 'fill-in':
182  if (type(operand[1]) == type('str')) and \
183  (operand[1].lower() == 'none'):
184  self.environment['fill-in'] = None
185  elif (type(operand[1]) == type('str')) and \
186  (operand[1].lower() != 'none'):
187  self.environment['fill-in'] = str(operand[1])
188  elif type(operand[1]) == type(1):
189  self.environment['fill-in'] = int(operand[1])
190  elif type(operand[1]) == type(1.1):
191  self.environment['fill-in'] = float(operand[1])
192  if op == 'parameter' and operand[1] != 'dataframe':
193  param_name = operand[1]
194  paramD_name = operand[2]
195  data = operand[3]
196  if paramD_name in self.session['parameters']:
197  self.session['parameters'][paramD_name][param_name] = data
198  if op == 'parameter' and operand[1] == 'dataframe':
199  paramD_name = operand[2]
200  df_name = operand[3]
201  if paramD_name in self.session['parameters']:
202  if df_name in self.session['MDF'].frames:
203  data = self.session['MDF'].frames[df_name]
204  self.session['parameters'][paramD_name]['dataframe'] = data
205  return None

575  def do_cast(self, operand):
576  '''!
577  Method to type cast one or more series in a data frame.
578 
579  Operand = [<data type to cast into>, <data frame name>,
580  [<series names>]]
581 
582  @param operand list: bytecode operand(s), if any
583  '''
584  if operand[0] == 'alpha': datatype = 'string'
585  if operand[0] == 'nonalpha': datatype = 'float'
586  if operand[0] == 'float': datatype = 'float'
587  if operand[0] == 'real': datatype = 'float'
588  if operand[0] == 'integer': datatype = 'integer'
589  df_name = operand[1]
590  series_names = operand[2]
591  if df_name not in self.session['MDF'].frames:
592  code = 'Error/006'
593  message = 'Dataframe name, %s, is not found' % (df_name)
594  self.error_message(code, message)
595  return None
596  df = self.session['MDF'].frames[df_name]
597  if series_names[0] == 'all':
598  df.cast(datatype, 'error_replace', 'all')
599  else:
600  error_sn = [s for s in series_names if s not in df.series_names]
601  if len(error_sn) > 0:
602  code = 'Warning/013'
603  message = 'The following series name(s) are not found and not casted: %s' \
604  % (' ,'.join(error_sn))
605  self.error_message(code, message)
606  series_names = [s for s in series_names if s in df.series_names]
607  for s in series_names:
608  df.cast(datatype, 'error_replace', s)
609  return None

1004  def interpret(self, statement):
1005  '''!
1006  Method to process an input statement by either sending it to
1007  Shell.intercept_processor(), for non-bytecodes statements, or
1008  Shell.command_processor(), for bytecodes statements after
1009  lexing and parsing the input statement into bytecodes.
1010 
1011  Bytecodes are returned by the corresponding language's lexer
1012  and parser in the form of a tuple, (<bytecode operator>,
1013  <bytecode operands>+), which is then split into operator (the
1014  byte operator) and a list of operands (the bytecode operands),
1015  if any.
1016 
1017  @param statement String: command-line statement
1018  '''
1019  try:
1020  self.history[str(self.count)] = statement
1021  if statement.lower() in ['copyright', 'copyright;',
1022  'credits', 'credits;',
1023  'exit', 'exit;',
1024  'license', 'license;',
1025  'quit', 'quit;',
1026  ]:
1027  state = self.intercept_processor(statement)
1028  if state == 'exit': return 'exit'
1029  else:
1030  bytecode = self.parser.parse(statement,
1031  self.environment['display_ast'])
1032  self.bytecode[str(self.count)] = bytecode
1033  operator = bytecode[0]
1034  if len(bytecode) == 1:
1035  operand = []
1036  else:
1037  operand = bytecode[1:]
1038  self.command_processor(operator, operand)
1039  self.count = self.count + 1
1040  except:
1041  error_message = list(self.formatExceptionInfo())
1042  for line in error_message:
1043  if (type(line) == list):
1044  for l in line:
1045  print(l)
